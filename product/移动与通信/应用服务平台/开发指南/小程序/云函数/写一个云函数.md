
### 用 callback 返回数据

```javascript
exports.main = (event, context, callback) => {
  setTimeout(() => {
    callback(null, {code: 0, data: 1});
  }, 200);
}
```

### 用 return 返回数据

* 直接返回
```javascript
exports.main = (event, context) => {
  
  return {code: 0, data: 1};
}
```

* `async/await`

```javascript
exports.main = async (event, context) => {
  
  let result = new Promise((resolve, reject) => {
      resolve({code: 0, data: 1});
  });
  return result;
}
```
